182 void updateMetrics() {
183 currentMetrics_.timestamp = pros::millis();
184 currentMetrics_.batteryVoltage = pros::battery::get_voltage() / 1000.0;
185 currentMetrics_.batteryCurrent = pros::battery::get_current() / 1000.0;
186
187 // Update motor health
188 currentMetrics_.motorHealth.clear();
189 for (const auto& motorPair : monitoredMotors_) {
190 if (motorPair.second) {
191 MotorHealth health;
192 health.port = motorPair.first;
193 health.current = motorPair.second->get_current_draw() / 1000.0;
194 health.temperature = motorPair.second->get_temperature();
195 health.power = motorPair.second->get_power();
196 health.rpm = motorPair.second->get_actual_velocity();
197 health.lastUpdate = currentMetrics_.timestamp;
198
199 // Stall detection
200 health.isStalled = (std::abs(health.power) > 0.1 && std::abs(health.rpm) < 5);
201
202 currentMetrics_.motorHealth.push_back(health);
203 }
204 }
205
206 // Maintain rolling history (last 50 readings = 5 seconds)
207 voltageHistory_.push_back(currentMetrics_.batteryVoltage);
208 currentHistory_.push_back(currentMetrics_.batteryCurrent);
209
210 if (voltageHistory_.size() > 50) voltageHistory_.pop_front();
211 if (currentHistory_.size() > 50) currentHistory_.pop_front();
212 }

214 void checkForIssues() {
215 std::vector<HealthIssue> newIssues;
216
217 // Check battery voltage (brownout detection)
218 if (currentMetrics_.batteryVoltage < 6.5) {
219 newIssues.push_back({
220 BROWNOUT,
221 currentMetrics_.batteryVoltage < 6.0 ? CRITICAL : WARNING,
222 "Low battery voltage detected",
223 currentMetrics_.timestamp,
224 "Check battery charge or reduce motor load"
225 });
226 }
227
228 // Check for voltage drops (tug of war detection)
229 if (voltageHistory_.size() >= 10) {
230 double avgVoltage = 0.0;
231 for (double v : voltageHistory_) avgVoltage += v;
232 avgVoltage /= voltageHistory_.size();
233
234 double voltageDrop = avgVoltage - currentMetrics_.batteryVoltage;
235 if (voltageDrop > 0.5) { // 0.5V drop indicates heavy load
236 newIssues.push_back({
237 TUG_OF_WAR,
238 WARNING,
239 "Sudden voltage drop detected - possible tug of war",
240 currentMetrics_.timestamp,
241 "Reduce motor power or check for external resistance"
242 });
243 }
244 }
245
246 // Check motor health
247 for (const auto& motor : currentMetrics_.motorHealth) {
248 // Overcurrent detection
249 if (motor.current > 2.5) { // 2.5A threshold
250 newIssues.push_back({
251 OVERCURRENT,
252 motor.current > 3.0 ? CRITICAL : WARNING,
253 "Motor " + std::to_string(motor.port) + " overcurrent",
254 currentMetrics_.timestamp,
255 "Check motor wiring and reduce load"
256 });
257 }
258
259 // Temperature warning
260 if (motor.temperature > 50.0) { // 50°C threshold
261 newIssues.push_back({
262 TEMPERATURE,
263 motor.temperature > 60.0 ? CRITICAL : WARNING,
264 "Motor " + std::to_string(motor.port) + " overheating",
265 currentMetrics_.timestamp,
266 "Allow motor to cool or reduce usage"
267 });
268 }
269
270 // Stall detection
271 if (motor.isStalled) {
272 newIssues.push_back({
273 BROKEN_DRIVE,
274 CRITICAL,
275 "Motor " + std::to_string(motor.port) + " appears stalled",
276 currentMetrics_.timestamp,
277 "Check motor connection and mechanical issues"
278 });
279 }
280 }
281
282 // Add new issues to the list
283 for (const auto& issue : newIssues) {
284 // Check if similar issue already exists
285 bool exists = false;
286 for (const auto& existing : issues_) {
287 if (existing.type == issue.type &&
288 existing.severity == issue.severity &&
289 (currentMetrics_.timestamp - existing.timestamp) < 5000) { // Within 5 seconds
290 exists = true;
291 break;
292 }
293 }
294
295 if (!exists) {
296 issues_.push_back(issue);
297 }
298 }
299
300 // Limit issue history
301 if (issues_.size() > 20) {
302 issues_.erase(issues_.begin(), issues_.begin() + issues_.size() - 20);
303 }
304
305 // Notify callback if health changed
306 if (!newIssues.empty() && healthCallback_) {
307 healthCallback_(getOverallHealth(), issues_);
308 }
309 }
