27 static double applyCurve(double input, CurveType type, double gain = 1.0) {
28 // Normalize input to -1.0 to 1.0
29 double normalized = input / 127.0;
30
31 // Apply deadzone
32 if (std::abs(normalized) < 0.05) {
33 return 0.0;
34 }
35
36 double curved = 0.0;
37
38 switch (type) {
39 case LINEAR:
40 curved = normalized;
41 break;
42
43 case EXPONENTIAL:
44 curved = expoCurve(normalized, gain);
45 break;
46
47 case SIGMOID:
48 curved = sigmoidCurve(normalized, gain);
49 break;
50
51 case CUBIC:
52 curved = cubicCurve(normalized, gain);
53 break;
54
55 case CHEESY_DRIVE:
56 curved = cheesyCurve(normalized, gain);
57 break;
58 }
59
60 // Denormalize back to -127 to 127
61 return curved * 127.0;
62 }
63
67 static double expoCurve(double input, double gain) {
68 return std::copysign(std::pow(std::abs(input), 1.0 + gain), input);
69 }
70
74 static double sigmoidCurve(double input, double gain) {
75 double scaled = input * gain * 3.0; // Scale for better curve shape
76 return std::tanh(scaled);
77 }
78
82 static double cubicCurve(double input, double gain) {
83 double absInput = std::abs(input);
84 double cubed = std::pow(absInput, 3.0);
85 return std::copysign(cubed + (absInput * gain), input);
86 }
87
91 static double cheesyCurve(double input, double gain) {
92 double absInput = std::abs(input);
93 // Mix of cubic for low speeds and linear for high speeds
94 double lowSpeed = std::pow(absInput, 2.5) * gain;
95 double highSpeed = absInput * (2.0 - gain);
96 double blended = absInput < 0.5 ? lowSpeed : highSpeed;
97 return std::copysign(blended, input);
98 }

147 void calculateDrivePowers(pros::Controller& controller, double& leftMotor, double& rightMotor) {
148 double throttle = 0.0;
149 double turn = 0.0;
150
151 // Get raw controller inputs
152 switch (config_.type) {
153 case ARCADE:
154 throttle = controller.get_analog(pros::E_CONTROLLER_ANALOG_LEFT_Y);
155 turn = controller.get_analog(pros::E_CONTROLLER_ANALOG_RIGHT_X);
156 break;
157
158 case TANK:
159 leftMotor = applyCurveToInput(controller.get_analog(pros::E_CONTROLLER_ANALOG_LEFT_Y), config_.throttleCurve, config_.throttleGain);
160 rightMotor = applyCurveToInput(controller.get_analog(pros::E_CONTROLLER_ANALOG_RIGHT_Y), config_.throttleCurve, config_.throttleGain);
161 return;
162
163 case CHEESY:
164 throttle = controller.get_analog(pros::E_CONTROLLER_ANALOG_LEFT_Y);
165 turn = controller.get_analog(pros::E_CONTROLLER_ANALOG_RIGHT_X);
166 // Cheesy Drive quick turn with right bumper
167 if (controller.get_digital(pros::E_CONTROLLER_DIGITAL_RIGHT)) {
168 turn = turn * 2.0; // Boost turn for quick turns
169 }
170 break;
171
172 case CURVATURE:
173 throttle = controller.get_analog(pros::E_CONTROLLER_ANALOG_LEFT_Y);
174 turn = controller.get_analog(pros::E_CONTROLLER_ANALOG_RIGHT_X);
175 // Curvature drive: turn controls radius, not speed
176 break;
177
178 case GTA:
179 // For VEX V5, shoulder buttons are digital, so we use them as on/off
180 double brake = controller.get_digital(pros::E_CONTROLLER_DIGITAL_L1) ? 127.0 : 0.0;
181 double accelerate = controller.get_digital(pros::E_CONTROLLER_DIGITAL_R1) ? 127.0 : 0.0;
182 throttle = accelerate - brake;
183 turn = controller.get_analog(pros::E_CONTROLLER_ANALOG_LEFT_X);
184 break;
185 }
186
187 // Apply curves
188 throttle = applyCurveToInput(throttle, config_.throttleCurve, config_.throttleGain);
189 turn = applyCurveToInput(turn, config_.turnCurve, config_.turnGain);
190
191 // Calculate motor powers based on drive type
192 switch (config_.type) {
193 case ARCADE:
194 leftMotor = throttle + turn;
195 rightMotor = throttle - turn;
196 break;
197
198 case CHEESY:
199 leftMotor = throttle + turn;
200 rightMotor = throttle - turn;
201 break;
202
203 case CURVATURE:
204 // Curvature drive implementation
205 if (throttle == 0.0) {
206 // Pure rotation when stopped
207 leftMotor = turn;
208 rightMotor = -turn;
209 } else {
210 // Curved movement
211 double radius = 1.0 / std::max(std::abs(turn), 0.01);
212 leftMotor = throttle * (radius + 1.0) / (2.0 * radius);
213 rightMotor = throttle * (radius - 1.0) / (2.0 * radius);
214 }
215 break;
216
217 case GTA:
218 leftMotor = throttle + turn;
219 rightMotor = throttle - turn;
220 break;
221
222 default:
223 leftMotor = 0.0;
224 rightMotor = 0.0;
225 break;
226 }
227
228 // Clamp outputs
229 leftMotor = std::max(-127.0, std::min(127.0, leftMotor));
230 rightMotor = std::max(-127.0, std::min(127.0, rightMotor));
231 }
