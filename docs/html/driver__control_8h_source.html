94 static double applyCurve(double input, CurveType type, double gain = 1.0) {
95 // Normalize input to -1.0 to 1.0
96 double normalized = input / 127.0;
97
98 // Apply deadzone
99 if (std::abs(normalized) < 0.05) {
100 return 0.0;
101 }
102
103 double curved = 0.0;
104
105 switch (type) {
106 case LINEAR:
107 curved = normalized;
108 break;
109
110 case EXPONENTIAL:
111 curved = expoCurve(normalized, gain);
112 break;
113
114 case SIGMOID:
115 curved = sigmoidCurve(normalized, gain);
116 break;
117
118 case CUBIC:
119 curved = cubicCurve(normalized, gain);
120 break;
121
122 case CHEESY_DRIVE:
123 curved = cheesyCurve(normalized, gain);
124 break;
125 }
126
127 // Denormalize back to -127 to 127
128 return curved * 127.0;
129 }

142 static double expoCurve(double input, double gain) {
143 return std::copysign(std::pow(std::abs(input), 1.0 + gain), input);
144 }
145
157 static double sigmoidCurve(double input, double gain) {
158 double scaled = input * gain * 3.0; // Scale for better curve shape
159 return std::tanh(scaled);
160 }
161
173 static double cubicCurve(double input, double gain) {
174 double absInput = std::abs(input);
175 double cubed = std::pow(absInput, 3.0);
176 return std::copysign(cubed + (absInput * gain), input);
177 }
178
190 static double cheesyCurve(double input, double gain) {
191 double absInput = std::abs(input);
192 // Mix of cubic for low speeds and linear for high speeds
193 double lowSpeed = std::pow(absInput, 2.5) * gain;
194 double highSpeed = absInput * (2.0 - gain);
195 double blended = absInput < 0.5 ? lowSpeed : highSpeed;
196 return std::copysign(blended, input);
197 }

305 void calculateDrivePowers(pros::Controller& controller, double& leftMotor, double& rightMotor) {
306 double throttle = 0.0;
307 double turn = 0.0;
308
309 // Get raw controller inputs
310 switch (config_.type) {
311 case ARCADE:
312 throttle = controller.get_analog(pros::E_CONTROLLER_ANALOG_LEFT_Y);
313 turn = controller.get_analog(pros::E_CONTROLLER_ANALOG_RIGHT_X);
314 break;
315
316 case TANK:
317 leftMotor = applyCurveToInput(controller.get_analog(pros::E_CONTROLLER_ANALOG_LEFT_Y), config_.throttleCurve, config_.throttleGain);
318 rightMotor = applyCurveToInput(controller.get_analog(pros::E_CONTROLLER_ANALOG_RIGHT_Y), config_.throttleCurve, config_.throttleGain);
319 return;
320
321 case CHEESY:
322 throttle = controller.get_analog(pros::E_CONTROLLER_ANALOG_LEFT_Y);
323 turn = controller.get_analog(pros::E_CONTROLLER_ANALOG_RIGHT_X);
324 // Cheesy Drive quick turn with right bumper
325 if (controller.get_digital(pros::E_CONTROLLER_DIGITAL_RIGHT)) {
326 turn = turn * 2.0; // Boost turn for quick turns
327 }
328 break;
329
330 case CURVATURE:
331 throttle = controller.get_analog(pros::E_CONTROLLER_ANALOG_LEFT_Y);
332 turn = controller.get_analog(pros::E_CONTROLLER_ANALOG_RIGHT_X);
333 // Curvature drive: turn controls radius, not speed
334 break;
335
336 case GTA:
337 // For VEX V5, shoulder buttons are digital, so we use them as on/off
338 double brake = controller.get_digital(pros::E_CONTROLLER_DIGITAL_L1) ? 127.0 : 0.0;
339 double accelerate = controller.get_digital(pros::E_CONTROLLER_DIGITAL_R1) ? 127.0 : 0.0;
340 throttle = accelerate - brake;
341 turn = controller.get_analog(pros::E_CONTROLLER_ANALOG_LEFT_X);
342 break;
343 }
344
345 // Apply curves
346 throttle = applyCurveToInput(throttle, config_.throttleCurve, config_.throttleGain);
347 turn = applyCurveToInput(turn, config_.turnCurve, config_.turnGain);
348
349 // Calculate motor powers based on drive type
350 switch (config_.type) {
351 case ARCADE:
352 leftMotor = throttle + turn;
353 rightMotor = throttle - turn;
354 break;
355
356 case CHEESY:
357 leftMotor = throttle + turn;
358 rightMotor = throttle - turn;
359 break;
360
361 case CURVATURE:
362 // Curvature drive implementation
363 if (throttle == 0.0) {
364 // Pure rotation when stopped
365 leftMotor = turn;
366 rightMotor = -turn;
367 } else {
368 // Curved movement
369 double radius = 1.0 / std::max(std::abs(turn), 0.01);
370 leftMotor = throttle * (radius + 1.0) / (2.0 * radius);
371 rightMotor = throttle * (radius - 1.0) / (2.0 * radius);
372 }
373 break;
374
375 case GTA:
376 leftMotor = throttle + turn;
377 rightMotor = throttle - turn;
378 break;
379
380 default:
381 leftMotor = 0.0;
382 rightMotor = 0.0;
383 break;
384 }
385
386 // Clamp outputs
387 leftMotor = std::max(-127.0, std::min(127.0, leftMotor));
388 rightMotor = std::max(-127.0, std::min(127.0, rightMotor));
389 }
