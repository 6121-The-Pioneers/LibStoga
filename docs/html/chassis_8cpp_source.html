92 void Chassis::moveToPoint(double X, double Y, unsigned int timeout, bool reverse) {
93 if (!odom || !lateral_control || !angular_control || !right || !left) return; // safety
94 ls::Timer timer(timeout);
95 ls::Position goal(X, Y, 0); // IGNORE THETA
96 unsigned int exit_timer = 0;
97 double initial_distance = odom->getPosition().distanceFromPoint(goal);
98 int log_counter = 0;
99
100 while (!timer.isDone()) {
101 if (enable_cancel && controller && controller->get_digital(cancel_button)) {
102 std::cout << "Motion cancelled by user" << std::endl;
103 break;
104 }
105 odom->compute();
106 ls::Position current_position = odom->getPosition();
107
108 // calculate amount of change:
109 double distance = goal.distanceFromPointSigned(current_position);
110 Angle theta_final = current_position.angleToPosition(goal);
111
112 if (reverse) {
113 theta_final += Angle(180);
114 theta_final = theta_final.normalize();
115 }
116 else {
117 distance *= -1;
118 }
119
120 double percent_complete = (1.0 - (fabs(distance) / initial_distance)) * 100.0;
121 if (static_cast<int>(percent_complete / 10) > log_counter) {
122 log_counter = static_cast<int>(percent_complete / 10);
123 std::cout << "Motion " << log_counter * 10 << "% complete" << std::endl;
124 }
125
126 Angle d_theta = current_position.theta.minimumAngleDifference(theta_final); // order might be subject to change
127
128 // calculate power and PID outputs:
129 double raw_power = lateral_control->update(distance);
130 double power = raw_power * move_priority(d_theta);
131 double turn = angular_control->update(d_theta.getAngle());
132
133 if (std::isnan(distance) || std::isnan(d_theta.getAngle())) {
134 DBG_PRINT(6, "NaN detected dist=%f ang=%f", distance, d_theta.getAngle());
135 break;
136 }
137 if (fabs(distance) <= threshold_lateral) {
138 exit_timer += LOOP_DELAY;
139 } else {
140 exit_timer = 0; // reset dwell timer when leaving threshold band
141 }
142
143 if (exit_timer > threshold_timeout) {
144 break;
145 }
146
147 // set power (clamped):
148 right->move(clampPower(power + turn)); // order might be subject to change
149 left->move(clampPower(power - turn)); // order might be subject to change
150
151 DBG_PRINT(5, "MT d=%.2f ang=%.2f p=%.2f t=%.2f", distance, d_theta.getAngle(), power, turn);
152
153 pros::delay(LOOP_DELAY);
154
155 }
156
157 right->move(0);
158 left->move(0);
159 }
