26 std::shared_ptr<Chassis> ChassisBuilder::build() {
27 // Motor groups
28 std::vector<pros::Motor> leftMotors;
29 for(int port : drivetrainConfig.leftMotors) {
30 leftMotors.emplace_back(port);
31 }
32 std::vector<pros::Motor> rightMotors;
33 for(int port : drivetrainConfig.rightMotors) {
34 rightMotors.emplace_back(port);
35 }
36
37 auto leftGroup = std::make_shared<pros::MotorGroup>(drivetrainConfig.leftMotors);
38 auto rightGroup = std::make_shared<pros::MotorGroup>(drivetrainConfig.rightMotors);
39
40 // Odom selection
41 std::shared_ptr<AbstractOdom> odom;
42 if (odomConfig.useThreeWheel) {
43 odom = std::make_shared<ThreeWheelOdom>(odomConfig.threeWheelParams);
44 } else {
45 odom = std::make_shared<ImuOdom>(odomConfig.imuParams);
46 }
47
48 // PIDs
49 auto drivePID = std::make_shared<PID>(controllerConfig.driveGains);
50 auto turnPID = std::make_shared<PID>(controllerConfig.turnGains);
51
52 // Chassis Config
53 ChassisConfig cc;
54 cc.turnSensitivity = controllerConfig.turnSensitivity;
55 cc.lateralThreshold = controllerConfig.tolerances.lateralThresh;
56 cc.angularThreshold = controllerConfig.tolerances.angularThresh;
57 cc.dwellTimeMs = controllerConfig.tolerances.dwellMs;
58 cc.enableAdaptivePID = controllerConfig.enableAdaptivePID;
59
60 // Create Chassis
61 auto chassis = std::make_shared<Chassis>(leftMotors, rightMotors, rightGroup, leftGroup, odom, drivePID, turnPID, cc);
62
63 // Note: PurePursuit is not part of the builder for now to keep it simple.
64 // It can be constructed separately if needed.
