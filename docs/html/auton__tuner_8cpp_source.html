16void AdaptivePIDController::update() {

19 // Only adapt periodically to avoid excessive computation
20 if (current_time - last_adaptation_time < ADAPTATION_INTERVAL) {

24
25 // Get current battery voltage

54void AdaptivePIDController::updateBatteryCompensation(double current_voltage) {
55 // Battery compensation: lower voltage = higher gains needed
56 float voltage_ratio = current_voltage / BATTERY_THRESHOLD_HIGH;
57
58 // More aggressive compensation for low battery
59 if (current_voltage < BATTERY_THRESHOLD_LOW) {
60 float low_battery_factor = BATTERY_THRESHOLD_LOW / current_voltage;
61 battery_compensation.kP = std::min(low_battery_factor * 1.3f, MAX_GAIN_MULTIPLIER);
62 battery_compensation.kI = std::min(low_battery_factor * 1.1f, MAX_GAIN_MULTIPLIER);
63 battery_compensation.kD = std::min(low_battery_factor * 1.2f, MAX_GAIN_MULTIPLIER);
64 } else {
65 // Gradual compensation for normal voltage range
66 battery_compensation.kP = std::max(voltage_ratio * 0.9f, 0.7f);
67 battery_compensation.kI = std::max(voltage_ratio * 0.95f, 0.8f);
68 battery_compensation.kD = std::max(voltage_ratio * 0.85f, 0.6f);
69 }
70
71 // Smooth voltage changes to prevent oscillations
72 static float last_p_comp = 1.0f, last_i_comp = 1.0f, last_d_comp = 1.0f;
73 battery_compensation.kP = last_p_comp * 0.8f + battery_compensation.kP * 0.2f;
74 battery_compensation.kI = last_i_comp * 0.8f + battery_compensation.kI * 0.2f;
75 battery_compensation.kD = last_d_comp * 0.8f + battery_compensation.kD * 0.2f;
76
77 last_p_comp = battery_compensation.kP;
78 last_i_comp = battery_compensation.kI;
79 last_d_comp = battery_compensation.kD;
80}

82void AdaptivePIDController::detectSurfaceConditions() {
83 if (recent_errors.size() < 5) return; // Need enough data
84
85 // Calculate average error and settling time
86 double avg_error = 0.0, avg_settling = 0.0;
87 for (size_t i = 0; i < recent_errors.size(); ++i) {
88 avg_error += recent_errors[i];
89 avg_settling += recent_settling_times[i];
90 }
91 avg_error /= recent_errors.size();
92 avg_settling /= recent_settling_times.size();
93
94 // Surface detection logic
95 SurfaceType detected_surface = UNKNOWN;
96
97 if (avg_error > 2.0 && avg_settling > 1500) {
98 detected_surface = ROUGH; // High error and long settling = rough surface
99 } else if (avg_error < 0.5 && avg_settling < 800) {
100 detected_surface = SMOOTH; // Low error and fast settling = smooth surface
101 } else {
102 detected_surface = CARPET; // Medium performance = carpet
103 }
104
105 // Only change surface type if consistently detected
106 if (detected_surface != current_surface) {
107 surface_detection_time = pros::millis();
108 } else if (pros::millis() - surface_detection_time > 3000) { // 3 seconds of consistent detection
109 if (current_surface != detected_surface) {
110 current_surface = detected_surface;
111 std::cout << "Surface changed to: " <<
112 (current_surface == SMOOTH ? "SMOOTH" :
113 current_surface == ROUGH ? "ROUGH" : "CARPET") << std::endl;
114 }
115 }
116}

118void AdaptivePIDController::updateSurfaceCompensation() {
119 // Surface compensation factors
120 switch (current_surface) {
121 case SMOOTH:
122 surface_compensation.kP = 0.9f; // Less aggressive on smooth surfaces
123 surface_compensation.kI = 1.1f; // More integral for steady state
124 surface_compensation.kD = 0.8f; // Less derivative on smooth
125 break;
126 case ROUGH:
127 surface_compensation.kP = 1.3f; // More aggressive on rough surfaces
128 surface_compensation.kI = 0.8f; // Less integral to avoid windup
129 surface_compensation.kD = 1.2f; // More derivative for rough
130 break;
131 case CARPET:
132 surface_compensation.kP = 1.0f; // Baseline for carpet
133 surface_compensation.kI = 1.0f;
134 surface_compensation.kD = 1.0f;
135 break;
136 default:
137 surface_compensation = {1.0f, 1.0f, 1.0f}; // No compensation for unknown
138 break;
139 }

142void AdaptivePIDController::applyAdaptiveGains() {
143 // Combine all compensation factors
144 PIDGains adapted_gains = base_gains;
145
146 adapted_gains.kP *= battery_compensation.kP * surface_compensation.kP;
147 adapted_gains.kI *= battery_compensation.kI * surface_compensation.kI;
148 adapted_gains.kD *= battery_compensation.kD * surface_compensation.kD;
149
150 // Apply safety limits
151 adapted_gains.kP = std::max(MIN_GAIN_MULTIPLIER * base_gains.kP,
152 std::min(MAX_GAIN_MULTIPLIER * base_gains.kP, adapted_gains.kP));
153 adapted_gains.kI = std::max(MIN_GAIN_MULTIPLIER * base_gains.kI,
154 std::min(MAX_GAIN_MULTIPLIER * base_gains.kI, adapted_gains.kI));
155 adapted_gains.kD = std::max(MIN_GAIN_MULTIPLIER * base_gains.kD,
156 std::min(MAX_GAIN_MULTIPLIER * base_gains.kD, adapted_gains.kD));
157
158 // Only update if gains changed significantly
159 const float GAIN_CHANGE_THRESHOLD = 0.05f; // 5% change threshold
160 if (fabs(adapted_gains.kP - current_gains.kP) > GAIN_CHANGE_THRESHOLD ||
161 fabs(adapted_gains.kI - current_gains.kI) > GAIN_CHANGE_THRESHOLD ||
162 fabs(adapted_gains.kD - current_gains.kD) > GAIN_CHANGE_THRESHOLD) {
163
164 current_gains = adapted_gains;
165 pid->setGains(current_gains);
166
167 // Optional: Log adaptation (can be commented out for performance)
168 // std::cout << "Adapted gains: P=" << current_gains.kP << ", I=" << current_gains.kI
169 // << ", D=" << current_gains.kD << std::endl;
170 }

177// Helper function to find ultimate gains using the safe Relay (Åström-Hägglund) method
178static std::pair<double, double> findUltimateGainsRelay(std::shared_ptr<Chassis> chassis, bool isAngular) {
179 const double relay_amplitude = 30.0; // Reduced from 60.0 to prevent excessive movement
180 const int min_cycles = 5; // Minimum number of relay cycles to observe
181 const int max_cycles = 20; // Maximum cycles to prevent infinite loop
182 const unsigned int cycle_timeout = 3000; // Reduced from 5000 to 3 seconds per cycle
183
184 std::shared_ptr<AbstractPID> pid = isAngular ? chassis->getAngularControl() : chassis->getLateralControl();
185 pid->setGains({0.0f, 0.0f, 0.0f}); // Disable PID, use pure relay control
186
187 std::vector<double> peaks;
188 std::vector<unsigned long> peak_times;
189 bool relay_state = true; // Start with positive relay output
190
191 chassis->getOdom()->resetAll();
192
193 for (int cycle = 0; cycle < max_cycles; ++cycle) {
194 ls::Timer cycle_timer(cycle_timeout);
195 double max_error = 0.0;
196 unsigned long max_error_time = 0;
197
198 while (!cycle_timer.isDone()) {
199 chassis->getOdom()->compute();
200 double current_pos = isAngular ? chassis->getOdom()->getAngle() : chassis->getOdom()->getPosition().Y;
201 double error = relay_state ? relay_amplitude : -relay_amplitude; // Relay control
202
203 if (fabs(current_pos) > fabs(max_error)) {
204 max_error = current_pos;
205 max_error_time = pros::millis();
206 }
207
208 double power = relay_state ? relay_amplitude : -relay_amplitude;
209 if (isAngular) {
210 chassis->arcade(0, power);
211 } else {
212 chassis->arcade(power, 0);
213 }
214
215 pros::delay(10);
216 }
217
218 peaks.push_back(max_error);
219 peak_times.push_back(max_error_time);
220 relay_state = !relay_state; // Switch relay state
221
222 if (peaks.size() >= min_cycles) {
223 // Check for stable oscillation
224 double avg_peak = 0.0;
225 for (double p : peaks) avg_peak += fabs(p);
226 avg_peak /= peaks.size();
227
228 if (peaks.size() >= 2) {
229 double period = (peak_times.back() - peak_times[peaks.size() - 2]) / 1000.0; // seconds
230 double amplitude = avg_peak;
231 double ku = 4 * relay_amplitude / (M_PI * amplitude); // Ultimate gain
232 double tu = period; // Ultimate period
233
234 std::cout << "Relay Method: Ku=" << ku << ", Tu=" << tu << "s" << std::endl;
235 return {ku, tu};
236 }
237 }
238 }
239
240 std::cout << "Relay Method failed to find stable oscillation." << std::endl;
241 return {0, 0};
242}

263 static bool y_pressed_last = false;
264 bool y_pressed = controller->get_digital(pros::E_CONTROLLER_DIGITAL_Y);

267 std::cout << "Tuning PAUSED. Press Y again to resume." << std::endl;

274 pros::delay(10);
275 }
276
277 // Wait for Y to be pressed again
278 while (!controller->get_digital(pros::E_CONTROLLER_DIGITAL_Y)) {
279 // Check for cancel with X button (require held for 500ms to prevent accidental presses)
280 static uint32_t x_press_start = 0;

307
308 // Check for cancel with X button (require held for 500ms to prevent accidental presses)
309 static uint32_t x_press_start_main = 0;

312

314 x_press_start_main = pros::millis();

331 // Generate new candidate gains by perturbing current best
332 PIDGains candidate_lateral = best_lateral;
333 PIDGains candidate_angular = best_angular;
334
335 // Random perturbation
336 candidate_lateral.kP *= (1.0f + (rand() % 21 - 10) * STEP_SIZE);

338 candidate_lateral.kD *= (1.0f + (rand() % 21 - 10) * STEP_SIZE);
339 candidate_angular.kP *= (1.0f + (rand() % 21 - 10) * STEP_SIZE);

341 candidate_angular.kD *= (1.0f + (rand() % 21 - 10) * STEP_SIZE);
342
343 // Ensure gains stay positive
344 candidate_lateral.kP = std::max(0.1f, candidate_lateral.kP);
345 candidate_lateral.kI = std::max(0.0f, candidate_lateral.kI);
346 candidate_lateral.kD = std::max(0.0f, candidate_lateral.kD);
347 candidate_angular.kP = std::max(0.1f, candidate_angular.kP);

391 }
392 }
393 }
394
395 switch (test) {
396 case 0: score += chassis->driveRelative(DISTANCE, 2000); break;
397 case 1: score += chassis->driveRelative(-DISTANCE, 2000); break;
398 case 2: score += chassis->turnToAngle(ANGLE, 2000); break;
399 case 3: score += chassis->turnToAngle(-ANGLE, 2000); break;
400 }
401 pros::delay(500); // Brief pause between tests
402 }
403
404 score /= num_tests;
405
406 std::cout << "Candidate score: " << score << " (best: " << best_score << ")" << std::endl;
407
408 // Update best if improved
409 if (score < best_score) {
410 best_score = score;
411 best_lateral = candidate_lateral;
412 best_angular = candidate_angular;
413 std::cout << "New best found!" << std::endl;
414 } else {
415 // Revert to best gains
416 chassis->getLateralControl()->setGains(best_lateral);
417 chassis->getAngularControl()->setGains(best_angular);
418 }
419
420 // Update LCD with current best
421 pros::lcd::print(0, "Iter: %d/%d", iter + 1, MAX_ITERATIONS);

424 pros::lcd::print(3, "A-P:%.2f I:%.2f D:%.3f", best_angular.kP, best_angular.kI, best_angular.kD);
425 }
426
427 // Set final best gains
428 lateral_gains = best_lateral;
429 angular_gains = best_angular;
430}

432void autoTune(std::shared_ptr<Chassis> chassis) {
433 std::cout << "Starting auto-tuning process..." << std::endl;
434
435 // Phase 1: Get baseline gains using Relay Method
436 std::cout << "Phase 1: Finding baseline gains with Relay Method..." << std::endl;
437
438 // Use pre-tuned angular values
439 PIDGains angular_gains = {1.74f, 3.16364f, 0.23925f};
440 chassis->getAngularControl()->setGains(angular_gains);
441
442 // Find lateral baseline using Relay Method
443 auto [lateral_ku, lateral_tu] = findUltimateGainsRelay(chassis, false);
444 if (lateral_ku == 0) {
445 std::cout << "Failed to find lateral baseline. Aborting." << std::endl;

450 PIDGains lateral_gains;
451 lateral_gains.kP = lateral_ku / 3.2;
452 lateral_gains.kI = lateral_ku / (2.2 * lateral_tu);
453 lateral_gains.kD = lateral_ku * lateral_tu / 6.3;
454 chassis->getLateralControl()->setGains(lateral_gains);
455
456 std::cout << "Baseline Lateral Gains: P=" << lateral_gains.kP << ", I=" << lateral_gains.kI << ", D=" << lateral_gains.kD << std::endl;
457
458 // Phase 2: Adaptive refinement
459 std::cout << "Phase 2: Adaptive refinement..." << std::endl;
460 pros::Controller* controller = chassis->getController();
461 adaptiveTune(chassis, lateral_gains, angular_gains, controller);
462
463 std::cout << "Auto-tuning complete!" << std::endl;
464 std::cout << "Final Angular Gains: P=" << angular_gains.kP << ", I=" << angular_gains.kI << ", D=" << angular_gains.kD << std::endl;
465 std::cout << "Final Lateral Gains: P=" << lateral_gains.kP << ", I=" << lateral_gains.kI << ", D=" << lateral_gains.kD << std::endl;
