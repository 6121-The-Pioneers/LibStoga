175 void ThreeWheelOdom::compute()
176 {
177 if(left) deltaL = left.get()->getLinearDeltaDistance();
178 if(right) deltaR = right.get()->getLinearDeltaDistance();
179 if(back) deltaB = back.get()->getLinearDeltaDistance();
180
181 Angle deltaAngle = getDeltaAngle();
182 double deltaAngleRad = deltaAngle.convertToRadians();
183
184 double localX = 0;
185 double localY = 0;
186
187 if (fabs(deltaAngleRad) < 1e-6) { // Linear approximation
188 localX = deltaB;
189 localY = deltaR;
190 } else { // Arc approximation
191 localX = 2 * sin(deltaAngleRad / 2.0) * (deltaB / deltaAngleRad + centerToBack);
192 localY = 2 * sin(deltaAngleRad / 2.0) * (deltaR / deltaAngleRad + centerToRight);
193 }
194
195 double avgAngle = this->pos.theta.convertToRadians() + (deltaAngleRad / 2.0);
196
197 this->pos.X += localX * cos(avgAngle) - localY * sin(avgAngle);
198 this->pos.Y += localX * sin(avgAngle) + localY * cos(avgAngle);
199 this->pos.theta += deltaAngle;
200 }

273 void ImuOdom::compute()
274 {
275 Angle deltaAngle = getDeltaAngle();
276 this->pos.theta += deltaAngle; // accumulate heading
277 Angle mathAngle = Angle(90 - this->pos.theta.normalize());
278 double dRad = deltaAngle.convertToRadians();
279 double relY = vert->getLinearDeltaDistance();
280 double relX = horiz->getLinearDeltaDistance();
281
282 if (fabs(dRad) < 1e-6) {
283 // straight approximation
284 deltaGlobalX = relX * cos(mathAngle.convertToRadians()) - relY * sin(mathAngle.convertToRadians());
285 deltaGlobalY = relY * cos(mathAngle.convertToRadians()) + relX * sin(mathAngle.convertToRadians());
286 } else {

289 double r1 = relX / dRad;
290 deltaGlobalX = r0 * sin(dRad) - r1 * (1 - cos(dRad));
291 deltaGlobalY = r1 * sin(dRad) + r0 * (1 - cos(dRad));
292 }
